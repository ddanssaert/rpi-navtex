};

export const resolveStation = (code) => {
  const name = STATION_NAMES[code];
  return name ? `${code} — ${name}` : `Unknown (${code})`;
};

export const resolveType = (code) => {
  const name = MESSAGE_TYPE_NAMES[code];
  return name ? `${code} — ${name}` : `Unknown (${code})`;
};
