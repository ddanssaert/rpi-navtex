   `import { useState, useEffect, useRef } from 'react';`

2. Add a module-level helper ABOVE the `const useFilters = () => {` line:
   ```js
   const syncFiltersToServer = async (filters) => {
       if (!('serviceWorker' in navigator)) return;
       const reg = await navigator.serviceWorker.ready;
       const sub = await reg.pushManager.getSubscription();
       if (!sub) return;
       await fetch('/push/filters', {
           method: 'PUT',
           headers: { 'Content-Type': 'application/json' },
           body: JSON.stringify({ endpoint: sub.endpoint, filters })
       });
   };
   ```
   (No-op when no service worker / no subscription. The backend only matters once a subscription exists.)

3. Inside the hook, AFTER the existing localStorage `useEffect`, add a debounced sync effect:
   ```js
   const syncDebounceRef = useRef(null);
   useEffect(() => {
       clearTimeout(syncDebounceRef.current);
       syncDebounceRef.current = setTimeout(() => {
           syncFiltersToServer(filters).catch(err =>
               console.warn('[useFilters] Push filter sync failed:', err)
           );
       }, 800);
       return () => clearTimeout(syncDebounceRef.current);
   }, [filters]);
   ```

4. Do NOT change `toggleStation`, `toggleType`, `isFiltered`, or the return object — public API stays identical.
